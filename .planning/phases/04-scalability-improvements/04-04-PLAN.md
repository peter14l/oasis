    ```dart
    import 'package:supabase_flutter/supabase_flutter.dart';
    import '../utils/image_utils.dart';
    
    class ImageService {
      final SupabaseClient _client;
      
      ImageService(this._client);
      
      /// Get optimized image URL from Supabase Storage
      String getOptimizedUrl({
        required String bucket,
        required String path,
        ImageTransform transform = const ImageTransform(),
      }) {
        // Use Supabase Storage transformation API
        final url = _client.storage
            .from(bucket)
            .getPublicUrl(path, transform: {
          if (transform.width != null) 'width': transform.width,
          if (transform.height != null) 'height': transform.height,
          'format': transform.format,
          'quality': transform.quality,
        });
        
        return url;
      }
      
      /// Convenience methods
      String getAvatar(String bucket, String userId) =>
          getOptimizedUrl(bucket: bucket, path: userId, transform: ImagePresets.avatar);
      
      String getThumbnail(String bucket, String path) =>
          getOptimizedUrl(bucket: bucket, path: path, transform: ImagePresets.thumbnail);
    }
    ```
