    ```dart
    class Cursor {
      final DateTime createdAt;
      final String id;
      
      const Cursor({required this.createdAt, required this.id});
      
      /// Serialize to API format: "2024-01-15T10:30:00Z_abc123"
      String toApiString();
      
      /// Deserialize from API format
      static Cursor fromApiString(String cursor);
      
      /// Empty cursor for first page
      static const Cursor empty = Cursor(createdAt: null, id: null);
      bool get isEmpty => createdAt == null;
    }
    
    /// Paginated response wrapper
    class PageResponse<T> {
      final List<T> items;
      final Cursor? nextCursor;
      final bool hasMore;
      
      const PageResponse({
        required this.items,
        this.nextCursor,
        required this.hasMore,
      });
    }
    ```

    ```dart
    /// Cursor-based pagination for Supabase
    class CursorPaginator {
      final SupabaseClient _client;
      final int defaultLimit;
      
      CursorPaginator(this._client, {this.defaultLimit = 20});
      
      /// Fetch a page using cursor
      Future<PageResponse<Map<String, dynamic>>> fetchPage({
        required String table,
        required String orderColumn,
        required String orderDirection, // 'desc' or 'asc'
        Cursor? cursor,
        int? limit,
        Map<String, dynamic>? filters,
      });
    }
    ```
    
    Key implementation:
    - Use `.gt()` with composite column: `where('created_at', cursor.createdAt).gt('id', cursor.id)`
    - For proper composite ordering, use: `.order(orderColumn, ascending: false).order('id', ascending: false)`
    - Return nextCursor from last item
