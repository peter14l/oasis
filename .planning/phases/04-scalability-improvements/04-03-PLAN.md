    ```dart
    import 'package:hive_flutter/hive_flutter.dart';
    import 'hive_boxes.dart';
    
    /// Generic cache service with offline support
    class CacheService<T> {
      final String boxName;
      final TypeAdapter<T>? adapter;
      
      Box<T>? _box;
      
      Future<Box<T>> get box async {
        _box ??= Hive.box<T>(boxName);
        return _box!;
      }
      
      /// Store item with optional TTL (in hours)
      Future<void> put(String key, T value, {int? ttlHours}) async {
        final b = await box;
        await b.put(key, value);
        if (ttlHours != null) {
          await b.put('${key}_ttl', DateTime.now().add(Duration(hours: ttlHours)));
        }
      }
      
      /// Get item, returns null if expired or not found
      Future<T?> get(String key) async {
        final b = await box;
        final ttl = b.get('${key}_ttl') as DateTime?;
        if (ttl != null && DateTime.now().isAfter(ttl)) {
          await b.delete(key);
          return null;
        }
        return b.get(key);
      }
      
      /// Get all items in cache
      Future<List<T>> getAll() async {
        final b = await box;
        return b.values.toList();
      }
      
      /// Delete item
      Future<void> delete(String key) async {
        final b = await box;
        await b.delete(key);
        await b.delete('${key}_ttl');
      }
      
      /// Clear all cache
      Future<void> clear() async {
        final b = await box;
        await b.clear();
      }
    }
    ```
